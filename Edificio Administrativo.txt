enable
configure terminal

hostname EdificioAdministrativo

!Configure el registro sincrónico para la línea de la consola
line console 0
logging synchronous
exit

banner motd #¡Advertencia! Solo personal autorizado.#

!Contraseña en modo privilegiado
enable secret PRIVTEC123

!Contraseña de consola
line console 0
password CONTEC123
login
exit

!Contraseña para acceso remoto
line vty 0 15
password REMTEC123
login
exit

!Todas las contraseñas deben estar encriptadas
service password-encryption

!Configure VLAN 
vlan 10
name Administracion
exit
vlan 20
name Tecnologia
exit
vlan 30
name Servers
exit
vlan 40
name Manufactura-Maestra
exit
vlan 50
name Manufactura-Garcia
exit
vlan 60
name Manufactura-Pesqueria
exit
vlan 100
name Red
exit

!Configure VLAN interface
interface vlan 10
ip address 192.168.10.1 255.255.255.0
no shutdown
exit

!Habilitar Enrutamiento
ip routing

!Configure Switch Ports

    !Acceso Capa 2 para VLANS 
interface Gig1/0/3
switchport mode trunk
switchport trunk encapsulation dot1q
switchport trunk allowed vlan 10,20,30,40,50,60,100
no shutdown
exit

interface Gig1/0/4
switchport mode trunk
switchport trunk encapsulation dot1q
switchport trunk allowed vlan 10,20,30,40,50,60,100
no shutdown
exit


    !Acceso Capa 3 Enrutamiento 

interface Gig1/0/1
no switchport
ip address 192.168.100.2 255.255.255.252
no shutdown
exit

interface Gig1/0/2
no switchport
ip address 192.168.100.6 255.255.255.252
no shutdown
exit


!OSPF
ROUTER OSPF 1
network 192.168.10.0 0.0.0.255 area 0
network 192.168.100.0 0.0.0.3 area 0
network 192.168.100.4 0.0.0.3 area 0
exit

!DCHP
!Exluir primera 5 direcciones
ip dhcp excluded-address 192.168.10.1 192.168.10.10

!Crear el pool DHCP 
ip dhcp pool INTEGRA

network 192.168.10.0 255.255.255.0
default-router 192.168.10.1
dns-server 192.168.30.2

!Save config
exit
